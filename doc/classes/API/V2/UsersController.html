def update
  user = current_user

  if user.update(allowed_params)
    render json: API::V2::UserSessionSerializer.new(user, serializer_params)
  else
    render_object_error object: user, serializer: API::V2::UserSessionSerializer, status: :unprocessable_entity
  end
end
